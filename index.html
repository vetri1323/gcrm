⚙️ Google Sheets Web App அமைவு

Apps Script Web App URL மூலம் Sheets-ல் நேரடியாக data சேமிக்கலாம். கீழே உள்ள படிகளை பின்பற்றவும்:

📋 ஒரு முறை அமைவு படிகள்:
1. உங்கள் Google Sheet திறக்கவும் → Extensions → Apps Script கிளிக்.
2. கீழே உள்ள Code-ஐ Apps Script Editor-ல் ஒட்டவும் (முழுமையாக மாற்றவும்).
3. Deploy → New Deployment → Web App → "Execute as: Me", "Who has access: Anyone" → Deploy.
4. உருவான Web App URL-ஐ கீழே ஒட்டவும்.
📄 Apps Script Code (Copy செய்து Editor-ல் ஒட்டவும்):
// Vehicle Management System - Google Apps Script
// Spreadsheet ID: 16RrGpYDvsgNlEsxKo8wjSvC6xIOu7s44A8W3EIn4RMk
const SPREADSHEET_ID = '16RrGpYDvsgNlEsxKo8wjSvC6xIOu7s44A8W3EIn4RMk';

const SHEETS = {
  VEHICLES: 'Vehicles',
  MAINTENANCE: 'Maintenance',
  REPORTS: 'Reports'
};

const VEHICLE_HEADERS   = ['ID','RegNo','Name','Type','Fuel','InsuranceExpiry','FCExpiry','Odometer','Status','Notes','Timestamp'];
const MAINT_HEADERS     = ['ID','Date','Month','VehicleID','VehicleReg','Category','Amount','NextServiceDate','Workshop','Status','Description','Timestamp'];
const REPORT_HEADERS    = ['ExportDate','FilterType','TotalExpense','VehicleCount','CompletedServices','PendingServices','GeneratedBy'];

function getOrCreateSheet(ss, name, headers) {
  let sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.appendRow(headers);
    sh.getRange(1, 1, 1, headers.length).setFontWeight('bold').setBackground('#1a73e8').setFontColor('#ffffff');
    sh.setFrozenRows(1);
  }
  return sh;
}

function doPost(e) {
  try {
    const raw = (e.parameter && e.parameter.payload) ? e.parameter.payload : e.postData.contents;
    const data = JSON.parse(raw);
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const action = data.action || 'append';

    if (action === 'append') {
      return handleAppend(ss, data);
    } else if (action === 'update') {
      return handleUpdate(ss, data);
    } else if (action === 'delete') {
      return handleDelete(ss, data);
    } else if (action === 'syncAll') {
      return handleSyncAll(ss, data);
    } else if (action === 'logReport') {
      return handleLogReport(ss, data);
    }
    return jsonResponse({status:'error', msg:'Unknown action'});
  } catch(err) {
    return jsonResponse({status:'error', msg: err.toString()});
  }
}

function doGet(e) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = e.parameter.sheet;
    if (!sheet) return jsonResponse({status:'ready', spreadsheetId: SPREADSHEET_ID});

    const headers = getHeaders(sheet);
    const sh = getOrCreateSheet(ss, sheet, headers);
    const rows = sh.getDataRange().getValues();
    if (rows.length <= 1) return jsonResponse({status:'ok', data:[], headers: rows[0]||[]});

    const keys = rows[0];
    const data = rows.slice(1).map(row => {
      const obj = {};
      keys.forEach((k,i) => obj[k] = row[i]);
      return obj;
    });
    return jsonResponse({status:'ok', data: data, count: data.length});
  } catch(err) {
    return jsonResponse({status:'error', msg: err.toString()});
  }
}

function handleAppend(ss, data) {
  const headers = getHeaders(data.sheet);
  const sh = getOrCreateSheet(ss, data.sheet, headers);
  sh.appendRow(data.row);
  return jsonResponse({status:'ok', action:'appended', sheet: data.sheet});
}

function handleUpdate(ss, data) {
  const headers = getHeaders(data.sheet);
  const sh = getOrCreateSheet(ss, data.sheet, headers);
  const rows = sh.getDataRange().getValues();
  const idCol = 0;
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][idCol]) === String(data.id)) {
      const range = sh.getRange(i + 1, 1, 1, data.row.length);
      range.setValues([data.row]);
      return jsonResponse({status:'ok', action:'updated', row: i+1});
    }
  }
  sh.appendRow(data.row);
  return jsonResponse({status:'ok', action:'inserted_on_update'});
}

function handleDelete(ss, data) {
  const headers = getHeaders(data.sheet);
  const sh = getOrCreateSheet(ss, data.sheet, headers);
  const rows = sh.getDataRange().getValues();
  const idCol = 0;
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][idCol]) === String(data.id)) {
      sh.deleteRow(i + 1);
      return jsonResponse({status:'ok', action:'deleted', row: i+1});
    }
  }
  return jsonResponse({status:'ok', action:'not_found'});
}

function handleSyncAll(ss, data) {
  const results = {};
  if (data.vehicles && Array.isArray(data.vehicles)) {
    const sh = getOrCreateSheet(ss, SHEETS.VEHICLES, VEHICLE_HEADERS);
    const last = sh.getLastRow();
    if (last > 1) sh.deleteRows(2, last - 1);
    data.vehicles.forEach(row => sh.appendRow(row));
    results.vehicles = data.vehicles.length;
  }
  if (data.maintenance && Array.isArray(data.maintenance)) {
    const sh = getOrCreateSheet(ss, SHEETS.MAINTENANCE, MAINT_HEADERS);
    const last = sh.getLastRow();
    if (last > 1) sh.deleteRows(2, last - 1);
    data.maintenance.forEach(row => sh.appendRow(row));
    results.maintenance = data.maintenance.length;
  }
  return jsonResponse({status:'ok', action:'syncAll', results});
}

function handleLogReport(ss, data) {
  const sh = getOrCreateSheet(ss, SHEETS.REPORTS, REPORT_HEADERS);
  sh.appendRow(data.row);
  return jsonResponse({status:'ok', action:'reportLogged'});
}

function getHeaders(sheet) {
  if (sheet === SHEETS.VEHICLES)   return VEHICLE_HEADERS;
  if (sheet === SHEETS.MAINTENANCE) return MAINT_HEADERS;
  if (sheet === SHEETS.REPORTS)    return REPORT_HEADERS;
  return [];
}

function jsonResponse(obj) {
  const output = ContentService.createTextOutput(JSON.stringify(obj));
  output.setMimeType(ContentService.MimeType.JSON);
  return output;
}
